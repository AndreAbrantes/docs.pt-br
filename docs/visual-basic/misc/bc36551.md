## <a name="example"></a><span data-ttu-id="dff5c-108">Exemplo</span><span class="sxs-lookup"><span data-stu-id="dff5c-108">Example</span></span>

Namespace StringUtility  
    <Extension()> _  
    Module StringExtensions  
        <Extension()> _

        End Sub  
    End Module  
End Namespace
