```xml

  <sources>  
      <source name="System.ServiceModel.MessageLogging">  
        <listeners>  
                 <add name="messages"

                 initializeData="c:\logs\messages.svclog" />  
          </listeners>  
      </source>

  <diagnostics>  
    <messageLogging

         logMalformedMessages="false"  
         logMessagesAtServiceLevel="true"

         maxMessagesToLog="3000"  
         maxSizeOfMessageToLog="2000"/>  
  </diagnostics>  
</system.serviceModel>  
```
