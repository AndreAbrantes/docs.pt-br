## <a name="enabling-tracing"></a><span data-ttu-id="00eee-118">Habilitar o rastreamento</span><span class="sxs-lookup"><span data-stu-id="00eee-118">Enabling Tracing</span></span>

```xml  
     <sources>  
          <source name="System.Transactions" switchValue="Warning">  
               <listeners>

                     type="System.Diagnostics.XmlWriterTraceListener"
                     initializeData= "tx.log" />  
               </listeners>  
          </source>  
     </sources>  
</system.diagnostics>  
</configuration>  
```
