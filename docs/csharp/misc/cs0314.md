## <a name="example"></a><span data-ttu-id="9e37c-108">Exemplo</span><span class="sxs-lookup"><span data-stu-id="9e37c-108">Example</span></span>  
 <span data-ttu-id="9e37c-109">O código a seguir gera CS0314:</span><span class="sxs-lookup"><span data-stu-id="9e37c-109">The following code generates CS0314:</span></span>

// cs0314.cs

public class ClassConstraint { }  
  
public class A<T> where T : ClassConstraint  
{ }  
  
public class B<T> : A<T> //CS0314  
{ }  
  
// Try using this instead.

{ }
