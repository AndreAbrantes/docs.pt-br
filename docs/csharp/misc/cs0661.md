```csharp  
// CS0661.cs

{  
   public static bool operator == (object o, Test t)  
   {  
      return true;  
   }  
  
   public static bool operator != (object o, Test t)  
   {  
      return true;  
   }  
  
   public override bool Equals(object o)  
   {  
      return true;  
   }  
  
   // uncomment the GetHashCode function to resolve  
   // public override int GetHashCode()  
   // {  
   //    return 0;  
   // }  
  
   public static void Main()  
   {  
   }  
}  
```
