## <a name="example"></a><span data-ttu-id="d103b-109">Exemplo</span><span class="sxs-lookup"><span data-stu-id="d103b-109">Example</span></span>

public static class Extensions  
{  
    public static S ExtMethod(this S s)

    }

{  
}
