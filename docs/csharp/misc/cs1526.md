```csharp

{  
   public static int i = 0;  
   public int myi = 0;  
}

{  
   public static void Main()  
   {  
      y py = new y;   // CS1526  
      y[] aoys = new y[10];   // Array of Ys

         aoys[i] = new y();   // an object of type y  
   }  
}  
```
