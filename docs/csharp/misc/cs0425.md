## <a name="example"></a><span data-ttu-id="6f3fa-107">Exemplo</span><span class="sxs-lookup"><span data-stu-id="6f3fa-107">Example</span></span>

```  
// CS0425.cs  
  
class C1  
{  
}  
  
class C2  
{  
}  
  
interface IBase  
{  
    void F<ItemType>(ItemType item) where ItemType : C1;  
}  
  
class Derived : IBase  
{  
    public void F<ItemType>(ItemType item) where ItemType : C2  // CS0425  
    {  
    }  
}

{  
    public static void Main()  
    {  
    }  
}  
```
