```csharp  
// CS0431.cs  
using A = Outer;  
  
public class Outer   
{  
   public class Inner   
   {  
      public static void Meth() {}  
   }  
}  
  
public class MyClass  
{  
   public static void Main()  
   {  
      A::Inner.Meth();   // CS0431  
      A.Inner.Meth();   // OK  
   }  
}  
```
