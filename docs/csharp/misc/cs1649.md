 <span data-ttu-id="d2e3b-110">O exemplo a seguir gera CS1649:</span><span class="sxs-lookup"><span data-stu-id="d2e3b-110">The following sample generates CS1649:</span></span>  
  
```csharp

    {  
        public int i;  
    }  
  
class Outer  
{  
    public readonly Inner inner = new Inner();  
}  
  
class D  
{  
    static void f(ref int iref)  
    {  
    }  
  
    static void Main()  
    {  
        Outer outer = new Outer();
        f(ref outer.inner.i);  // CS1649  
        // Try this code instead:  
        // int tmp = outer.inner.i;  
        // f(ref tmp);  
    }  
}  
```
