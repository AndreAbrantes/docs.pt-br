 <span data-ttu-id="f108b-108">O exemplo a seguir gera CS0415:</span><span class="sxs-lookup"><span data-stu-id="f108b-108">The following sample generates CS0415:</span></span>  
  
```csharp
// CS0415.cs  
using System;  
using System.Runtime.CompilerServices;  
  
public interface IA  
{  
    int this[int index]  
    {  
        get;  
        set;  
    }  
}  
  
public class A : IA  
{  
    [IndexerName("Item")]  // CS0415  
    int IA.this[int index]  
    // Try this line instead:  
    // public int this[int index]  
    {  
        get { return 0; }  
        set { }  
    }  
  
    static void Main()  
    {  
    }  
}  
```
