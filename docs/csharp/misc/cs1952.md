## <a name="example"></a><span data-ttu-id="4415e-107">Exemplo</span><span class="sxs-lookup"><span data-stu-id="4415e-107">Example</span></span>  
 <span data-ttu-id="4415e-108">O código a seguir produz CS1952:</span><span class="sxs-lookup"><span data-stu-id="4415e-108">The following code produces CS1952:</span></span>

using System.Linq.Expressions;  
  
class Test  
{  
    public static int M(__arglist)

        return 1;  
    }

        Expression<Func<int, int>> f = x => Test.M(__arglist(x)); // CS1952

    }  
}
