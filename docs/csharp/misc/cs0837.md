## <a name="example"></a><span data-ttu-id="02204-109">Exemplo</span><span class="sxs-lookup"><span data-stu-id="02204-109">Example</span></span>

    public delegate void Del();

    {
        static int Main()

            bool b1 = (() => { }) is Del;   // CS0837

        }

}
