 <span data-ttu-id="b40b4-104">Rotinas de conversão duplicados não são permitidas em uma classe.</span><span class="sxs-lookup"><span data-stu-id="b40b4-104">Duplicate conversion routines are not allowed in a class.</span></span>

    public class ii  
    {  
        public class iii  
        {  
        public static implicit operator int(iii aa)

        {  
        }  
    }  
}
