```  
// CS0152.cs  
namespace MyNamespace  
{  
   public class MyClass  
   {  
      public static void Main()  
      {  
         int i = 0;  
  
         switch (i)  
         {  
            case 1:  
               i++;  
               return;  
  
            case 1:   // CS0152, two case 1 statements

               return;  
         }  
      }  
   }  
}
```
