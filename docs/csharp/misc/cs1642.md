## <a name="example"></a><span data-ttu-id="14e25-106">Exemplo</span><span class="sxs-lookup"><span data-stu-id="14e25-106">Example</span></span>

{  
   fixed int a[10];   // CS1642

    fixed int a[10];  
}

{  
   public int[] a = null;  
}
