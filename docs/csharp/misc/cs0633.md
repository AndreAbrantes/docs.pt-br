## <a name="example"></a><span data-ttu-id="dad51-109">Exemplo</span><span class="sxs-lookup"><span data-stu-id="dad51-109">Example</span></span>

{  
   [IndexerName("Invalid+Identifier")]   // CS0633

   {

   }
