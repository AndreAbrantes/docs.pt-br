 <span data-ttu-id="7f678-107">O exemplo a seguir gera CS0183:</span><span class="sxs-lookup"><span data-stu-id="7f678-107">The following sample generates CS0183:</span></span>  
  
```csharp  
// CS0183.cs  
// compile with: /W:1  
using System;  
public class Test  
{  
   public static void F(Int32 i32, String str)  
   {  
      if (str is Object)          // OK  
         Console.WriteLine( "str is an object" );  
      else  
         Console.WriteLine( "str is not an object" );  
  
      if (i32 is Object)   // CS0183  
         Console.WriteLine( "i32 is an object" );  
      else  
         Console.WriteLine( "i32 is not an object" ); // never reached  
   }  
  
   public static void Main()  
   {  
  
      F(0, "CS0183");  
      F(120, null);   
   }  
}  
```
