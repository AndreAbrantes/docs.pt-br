## <a name="example"></a><span data-ttu-id="a4ffd-107">Exemplo</span><span class="sxs-lookup"><span data-stu-id="a4ffd-107">Example</span></span>  
 <span data-ttu-id="a4ffd-108">O código a seguir gera CS0754:</span><span class="sxs-lookup"><span data-stu-id="a4ffd-108">The following code generates CS0754:</span></span>

    public interface IF  
    {  
        void Part();  
    }  
  
    public partial class C : IF  
    {  
        partial void IF.Part(); //CS0754

        {  
            return 1;  
        }  
    }
