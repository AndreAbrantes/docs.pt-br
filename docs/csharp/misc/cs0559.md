   public static implicit operator int(iii x)

      return null;  
   }

   }

 <span data-ttu-id="46b16-110">O exemplo a seguir gera CS0559.</span><span class="sxs-lookup"><span data-stu-id="46b16-110">The following sample generates CS0559.</span></span>  
  
```csharp  
// CS0559_b.cs  
// compile with: /target:library  
public struct S  
{  
   public static S operator ++(S? s) { return new S(); }   // CS0559

{  
// OK  
   public static T operator --(T t) { return new T(); }  
   public static T operator ++(T t) { return new T(); }  
  
   public static T? operator --(T? t) { return new T(); }  
   public static T? operator ++(T? t) { return new T(); }  
}  
```
