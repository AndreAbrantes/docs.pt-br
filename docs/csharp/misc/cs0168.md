<span data-ttu-id="008d0-105">O exemplo a seguir gera um aviso CS0168:</span><span class="sxs-lookup"><span data-stu-id="008d0-105">The following sample generates a CS0168 warning:</span></span>

public class clx
{
    public int i;
}

public class clz
{
    public static void Main() {
        clx a; // CS0168, the variable 'a' is declared but never used
        // try the following line instead
        // clx a = new clx();  // this does not generate a warning because the clx constructor must execute.
    }
}
