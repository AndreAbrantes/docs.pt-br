 <span data-ttu-id="eb705-107">O exemplo a seguir gera CS0271:</span><span class="sxs-lookup"><span data-stu-id="eb705-107">The following example generates CS0271:</span></span>  
  
```  
// CS0271.cs  
public class MyClass  
{  
   public int Property  
   {  
      private get { return 0; }  
      set { }  
   }  
  
   public int Property2  
   {  
      get { return 0; }  
      set { }  
   }  
}  
  
public class Test  
{  
   public static void Main(string[] args)   
   {  
      MyClass c = new MyClass();  
      int a = c.Property;   // CS0271  
      int b = c.Property2;   // OK  
   }  
}  
```
