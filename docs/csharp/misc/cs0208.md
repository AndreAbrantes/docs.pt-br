## <a name="example"></a><span data-ttu-id="f2e82-109">Exemplo</span><span class="sxs-lookup"><span data-stu-id="f2e82-109">Example</span></span>  
 <span data-ttu-id="f2e82-110">O exemplo a seguir gera CS0208:</span><span class="sxs-lookup"><span data-stu-id="f2e82-110">The following sample generates CS0208:</span></span>  
  
```  
// CS0208.cs  
// compile with: /unsafe  
  
class myClass  
{  
    public int a = 98;  
}  
  
struct myProblemStruct  
{  
    string s;  
    float f;  
}  
  
struct myGoodStruct  
{  
    int i;  
    float f;  
}  
  
public class MyClass  
{  
    unsafe public static void Main()  
    {  
        // myClass is a class, a managed type.  
        myClass s = new myClass();    
        myClass* s2 = &s;    // CS0208  
  
        // The struct contains a string, a managed type.  
        int i = sizeof(myProblemStruct); //CS0208  
  
        // The struct contains only value types.  
        i = sizeof(myGoodStruct); //OK  
  
    }  
}  
```
