## <a name="example"></a><span data-ttu-id="8738a-110">Exemplo</span><span class="sxs-lookup"><span data-stu-id="8738a-110">Example</span></span>

{  
    public void Dispose() {  }  
}

{  
    public static void dummy()  
    {  
        ValidBase vb = null;  
        using (vb)

        vb = null;  
    }

}
