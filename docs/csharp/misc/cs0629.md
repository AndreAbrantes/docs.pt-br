 <span data-ttu-id="8690f-104">O [condicional](http://msdn.microsoft.com/en-us/e1c4913b-74d0-421a-8a6d-c14b3f0e68fb) atributo não pode ser usado na implementação de uma interface.</span><span class="sxs-lookup"><span data-stu-id="8690f-104">The [Conditional](http://msdn.microsoft.com/en-us/e1c4913b-74d0-421a-8a6d-c14b3f0e68fb) attribute cannot be used on the implementation of an interface.</span></span>

```  
// CS0629.cs  
interface MyInterface  
{  
   void MyMethod();  
}  
  
public class MyClass : MyInterface  
{  
   [System.Diagnostics.Conditional("debug")]  
   public void MyMethod()    // CS0629, remove the Conditional attribute  
   {  
   }

   {  
   }  
}  
```
