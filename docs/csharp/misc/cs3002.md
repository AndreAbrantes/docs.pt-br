## <a name="example"></a><span data-ttu-id="a9e24-106">Exemplo</span><span class="sxs-lookup"><span data-stu-id="a9e24-106">Example</span></span>

```csharp  
// CS3002.cs

public class a  
{  
    public ushort bad()   // CS3002, public method  
    {  
        ushort a;  
        a = ushort.MaxValue;  
        return a;  
    }

    {  
        ushort a;  
        a = ushort.MaxValue;  
        return a;  
    }  
  
    public static void Main()  
    {  
    }  
}  
```
