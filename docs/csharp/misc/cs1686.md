## <a name="example"></a><span data-ttu-id="ddecf-105">Exemplo</span><span class="sxs-lookup"><span data-stu-id="ddecf-105">Example</span></span>

```

{  
   public unsafe delegate int * MyDelegate();  
  
   public unsafe int * Test()

      int j = 0;  
      MyDelegate d = delegate { return &j; };   // CS1686  
      return &j;   // OK  
   }  
}
```
