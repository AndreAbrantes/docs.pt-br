## <a name="example"></a><span data-ttu-id="90669-109">Exemplo</span><span class="sxs-lookup"><span data-stu-id="90669-109">Example</span></span>

class Test  
{  
    static void Main()  
    {  
        int[] array = new int[] { 1, 2, 3, 4, 5 };
