## <a name="example"></a><span data-ttu-id="e56cc-109">Exemplo</span><span class="sxs-lookup"><span data-stu-id="e56cc-109">Example</span></span>

{  
    static void Main()  
    {  
        int[] array = new int[] { 1, 2, 3, 4, 5 };
