## <a name="example"></a><span data-ttu-id="eb3cc-107">Exemplo</span><span class="sxs-lookup"><span data-stu-id="eb3cc-107">Example</span></span>

```  
// CS0241.cs  
public class A  
{  
   public void Test(int i = 9) {}   // CS0241  
}  
  
public class B  
{  
   public void Test() { Test(9); }  
   public void Test(int i)  {}  
}  
  
public class C  
{  
   public static void Main()  
   {   
      B x = new B();  
      x.Test();  
   }  
}  
```
