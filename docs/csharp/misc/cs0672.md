 <span data-ttu-id="45440-109">O exemplo a seguir gera CS0672:</span><span class="sxs-lookup"><span data-stu-id="45440-109">The following sample generates CS0672:</span></span>  
  
```csharp  
// CS0672.cs  
// compile with: /W:1  
class MyClass  
{  
   [System.Obsolete]  
   public virtual void ObsoleteMethod()  
   {  
   }  
}  
  
class MyClass2 : MyClass  
{  
   public override void ObsoleteMethod()   // CS0672  
   {  
   }  
}  
  
class MainClass  
{  
   static public void Main()  
   {  
   }  
}  
```
