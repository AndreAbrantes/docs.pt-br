```csharp  
   public static MyClass operator ++ (MyClass c)  
   {

         return 1;  
      }  
      set  
      {  
      }  
   }

   {  
      op_Increment(null);   // CS0571  
      // use the increment operator as follows  
      // MyClass x = new MyClass();  
      // x++;

      // try the following line instead  
      // prop = 1;  
   }  
}  
```
