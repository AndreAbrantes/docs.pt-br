## <a name="example"></a><span data-ttu-id="1c6fa-115">Exemplo</span><span class="sxs-lookup"><span data-stu-id="1c6fa-115">Example</span></span>

   interface I  
   {  
      void m();  
      static public void f();   // CS0106
