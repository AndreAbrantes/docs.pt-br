## <a name="example"></a><span data-ttu-id="6c3af-109">Exemplo</span><span class="sxs-lookup"><span data-stu-id="6c3af-109">Example</span></span>

```cpp  
// compile with: /clr /LD  
using namespace System;  
public ref class MCPP  
{  
public:  
    property int Prop [int,int]  
    {  
        int get( int i, int b )  
        {  
            return i;  
        }  
    }  
};  
```
